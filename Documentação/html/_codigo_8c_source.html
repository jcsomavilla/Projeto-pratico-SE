117void task_lcd( void *pvParameters ){
118 float distancia = 0.0;
119 int leitura_MQ2 = 0;
120 char linha_str[16] = {0x00}; //formata a linha a ser escrita no display
121 int distancia_cm = 0; //mostra somente a parte inteira da distancia, porém o decimal é transportado para fila
122
123 while(1){
124 // xQueuePeek -> "espia" a fila do sensor
125 if( xQueuePeek(fila_ultrassonico, &distancia, TIMER_QUEUE_WAIT) ){ //escreve a ultima leitura do sensor
126 lcd.setCursor(0,0); //posiciono no começo da linha no display
127 lcd.print(LCD_16X2_CLEAN_LINE); //escrevo uma linha "em branco"
128 lcd.setCursor(0,0); //reposiciono no começo da linha
129
130 distancia_cm = (int)distancia;
131 sprintf (linha_str, "Dist: %d cm", distancia_cm); //formato a escrita no displaY
132 lcd.print(linha_str);
133 }
134
135
136 if( xQueuePeek(fila_MQ2, &leitura_MQ2, TIMER_QUEUE_WAIT) ){
137 lcd.setCursor(0,1);
138 lcd.print(LCD_16X2_CLEAN_LINE);
139 lcd.setCursor(0,1);
140
141 sprintf (linha_str, "MQ2: %d", leitura_MQ2);
142 lcd.print(linha_str);
143 }
144
145
146 vTaskDelay( LCD_TIMER_VERIFIC / portTICK_PERIOD_MS ); // tempo de verificação de atualização do display
147 // porTICK_PERIOD_MS converte o tempo setado em ms no LCD_TIMER_VERIFIC em ticks de processador.
148 }
149}

151void task_MQ2( void *pvParameters ){
152 int leitura_analogica = 0;
153 int Pinbuzzer = 8; //PINO UTILIZADO PELO BUZZER
154 int pinled = 7; //PINO UTILIZADO PELO LED
155
156 int leitura_sensor = 400;//DEFININDO UM VALOR LIMITE (NÍVEL DE GÁS NORMAL)
157
158 while(1){
159 leitura_analogica = analogRead(ANALOG_A0);
160
161 //Insere leitura na fila
162 xQueueOverwrite(fila_MQ2, (void *)&leitura_analogica);
163
164 /*escreve a leiturana serial. Tentativa de controle do semafoto é feita
165 até o tempo definido em TIME_SEMPH_WAIT*/
166
167 if ( xSemaphoreTake(semaforo_serial, TIMER_SEMPH_WAIT ) == pdTRUE ){
168 Serial.print("- Leitura MQ-2: ");
169 Serial.println(leitura_analogica);
170 xSemaphoreGive(semaforo_serial);
171 }
172
173 vTaskDelay( MQ2_TIMER_LEITURA/ portTICK_PERIOD_MS ); //aguarda tempo determinado em MQ2_TIMER_LEITURA para realizar prox leitura;
174
175 pinMode(ANALOG_A0, INPUT); //DEFINE O PINO COMO ENTRADA
176 pinMode(Pinbuzzer, OUTPUT); //DEFINE O PINO COMO SAÍDA
177 pinMode(pinled, OUTPUT); //DEFINE O PINO COMO SAIDA
178 Serial.begin(115200);//INICIALIZA A SERIAL
179
180 int valor_analogico = analogRead(ANALOG_A0); //VARIÁVEL RECEBE O VALOR LIDO NO PINO ANALÓGICO
181
182 if (valor_analogico > leitura_sensor){//SE VALOR LIDO NO PINO ANALÓGICO FOR MAIOR QUE O VALOR LIMITE, FAZ
183 digitalWrite(Pinbuzzer, HIGH); //ATIVA O BUZZER E O MESMO EMITE O SINAL SONORO
184 digitalWrite(pinled, HIGH); //ativa o led
185 }
186 else{ //SENÃO, FAZ
187 digitalWrite(Pinbuzzer, LOW);//BUZZER DESLIGADO
188 digitalWrite(pinled, LOW);//LED DESLIGADO
189 }
190 delay(100); //INTERVALO DE 100 MILISSEGUNDOS
191 }
192
193}
